      1 // services/FoodAI.ts (núcleo del sistema)
      2 
      3 async analyzeFood(input: string): Promise<FoodAnalysisResult> {
      4     // ESTRATEGIA EN CASCADA (transparente para el usuario)
      5     
      6     // 1. Caché (instantáneo)
      7     const cached = await this.cache.get(input);
      8     if (cached) return cached;
      9 
     10     // 2. Match exacto BD local (rápido)
     11     const dbMatch = await this.searchDatabase(input);
     12     if (dbMatch && dbMatch.confidence > 0.9) {
     13         return dbMatch; // Sin IA, rápido
     14     }
     15 
     16     // 3. Reglas heurísticas (muy rápido)
     17     const heuristic = this.heuristicMatch(input);
     18     if (heuristic && heuristic.confidence > 0.75) {
     19         return heuristic; // Sin IA
     20     }
     21 
     22     // 4. IA solo si es necesario (lento, último recurso)
     23     const deviceTier = await this.detectDeviceTier();
     24     if (deviceTier === 'low') {
     25         return this.fallbackEstimate(input); // Sin IA en gama baja
     26     }
     27 
     28     return await this.queryAI(input); // IA Qwen2.5
     29 }
